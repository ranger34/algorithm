Determine whether an integer is a palindrome. Do this without extra space.

class Solution {
public:
    bool isPalindrome(int x) {
        if(x < 0) return false;
        int rev = 0;
        int tmp = x;
        while(x != 0)
        {
            rev = rev * 10 + x % 10;
            x = x / 10;
        }
        if(rev == tmp)
            return true;
        else
            return false;
    }
};
